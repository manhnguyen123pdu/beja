1.1. Vấn đề: Code "truyền thống" hỗn độn

Khi chưa biết MVC, sinh viên thường viết như thế này:

@WebServlet("/users")
public class UserServlet extends HttpServlet {
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
        throws IOException {
        
        // 1. Code kết nối DB trong Servlet
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection conn = DriverManager.getConnection(
                "jdbc:mysql://localhost:3306/testdb", "root", "123456");

            Statement stmt = conn.createStatement();

            }
            
            resp.getWriter().println("</table></body></html>");
            
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}

CHƯƠNG 2: MODEL - PHẦN 1 (CLASS ĐẠI DIỆN DỮ LIỆU)

2.1. Khái niệm

Model trong MVC là thành phần đại diện cho dữ liệu của ứng dụng.
Có 2 loại Model:

  • Entity Model (Class ánh xạ bảng DB) - Học trong chương này
  • DAO Model (Class truy vấn DB) - Học trong chương 3

2.2. Entity Model là gì?

Định nghĩa: Một class Java bình thường, mỗi thuộc tính tương ứng với một cột trong bảng database.

2.3. Cú pháp tạo Entity Model

Java - Cú pháp cơ bản

CHƯƠNG 3: MODEL - PHẦN 2 (DAO - TRUY VẤN DATABASE)

3.1. DAO là gì?

DAO = Data Access Object (Đối tượng truy cập dữ liệu)

Định nghĩa: Một class chứa các phương thức để thao tác với database (SELECT, INSERT, UPDATE, DELETE).

Tại sao gọi là Model? Vì DAO nằm trong tầng Model của MVC - nó chịu trách nhiệm cung cấp dữ liệu.

3.2. Cấu trúc một class DAO chuẩn

Java - Cấu trúc DAO
package dao;

import model.TenModel;
import util.DBConnection;
import java.sql.*;
import java.util.*;

public class TenDAO {
    
    // 1. READ ALL - Lấy tất cả bản ghi
    public List<TenModel> getAll() {
        List<TenModel> list = new ArrayList<>();
        String sql = "SELECT * FROM ten_bang";
        
        try (Connection conn = DBConnection.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {
            
            while (rs.next()) {
                TenModel obj = new TenModel();
                obj.setId(rs.getInt("id"));
                obj.setTen(rs.getString("ten"));
                list.add(obj);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return list;
    }
    
    // 2. READ BY ID - Lấy 1 bản ghi theo id
    public TenModel getById(int id) {
        String sql = "SELECT * FROM ten_bang WHERE id = ?";
        
        try (Connection conn = DBConnection.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setInt(1, id);
            ResultSet rs = pstmt.executeQuery();
            
            if (rs.next()) {
                TenModel obj = new TenModel();
                obj.setId(rs.getInt("id"));
                obj.setTen(rs.getString("ten"));
                return obj;
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return null;
    }
    
    // 3. CREATE - Thêm mới
    public boolean insert(TenModel obj) {
        String sql = "INSERT INTO ten_bang (ten) VALUES (?)";
        
        try (Connection conn = DBConnection.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setString(1, obj.getTen());
            return pstmt.executeUpdate() > 0;
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }
    
    // 4. UPDATE - Cập nhật
    public boolean update(TenModel obj) {
        String sql = "UPDATE ten_bang SET ten = ? WHERE id = ?";
        
        try (Connection conn = DBConnection.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setString(1, obj.getTen());
            pstmt.setInt(2, obj.getId());
            return pstmt.executeUpdate() > 0;
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }
    
    // 5. DELETE - Xóa
    public boolean delete(int id) {
        String sql = "DELETE FROM ten_bang WHERE id = ?";
        
        try (Connection conn = DBConnection.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            pstmt.setInt(1, id);
            return pstmt.executeUpdate() > 0;
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }
}

3.3. DBConnection.java - Class tiện ích kết nối database

util/DBConnection.java
package util;

import java.sql.*;

public class DBConnection {
    // Thông tin kết nối - CẦN SỬA THEO MÁY CỦA BẠN
    private static final String URL = "jdbc:mysql://localhost:3306/ten_database?useSSL=false&serverTimezone=UTC";
    private static final String USER = "root"; // tên đăng nhập MySQL
    private static final String PASSWORD = "123456"; // mật khẩu MySQL
    
    // Khối static - chạy 1 lần duy nhất khi class được load
    static {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            System.out.println("MySQL Driver loaded successfully!");
        } catch (ClassNotFoundException e) {
            System.out.println("Lỗi: Không tìm thấy driver MySQL");
            e.printStackTrace();
        }
    }
    
    // Phương thức trả về Connection
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USER, PASSWORD);
    }
    
    // Test kết nối (dùng để kiểm tra)
    public static void main(String[] args) {
        try {
            Connection conn = getConnection();
            System.out.println("Kết nối database thành công!");
            conn.close();
        } catch (SQLException e) {
            System.out.println("Kết nối database thất bại: " + e.getMessage());
        }
    }
}

CHƯƠNG 4: VIEW - JSP HIỂN THỊ DỮ LIỆU

4.1. View là gì?

View trong MVC là phần giao diện người dùng. Trong Java Web, View thường là các file JSP (JavaServer Pages).

  • Nhiệm vụ của View: Hiển thị dữ liệu nhận được từ Controller, Gửi request lên Controller (qua form, link, button)
  • JSP không được chứa: Logic xử lý phức tạp, Code kết nối database, Code SQL

4.2. Cú pháp JSP cơ bản

Directive (Chỉ thị)

JSP
<%@ page contentType="text/html;charset=UTF-8" language="java" %>
<%@ taglib uri="http://java.sun.com/jsp/jstl/core" prefix="c" %>

Expression Language (EL) - ★ NÊN DÙNG

JSP - EL
<p>${userName}</p> <%-- Hiển thị biến userName --%>
<p>${user.fullName}</p> <%-- Hiển thị thuộc tính fullName của object user --%>
<p>${userList[0].name}</p> <%-- Hiển thị phần tử đầu tiên trong List --%>

Các thẻ JSTL thông dụng:

CHƯƠNG 5: CONTROLLER - SERVLET ĐIỀU KHIỂN

5.1. Controller là gì?

Controller trong MVC là thành phần điều phối toàn bộ ứng dụng.

Nhiệm vụ chính:

  • Nhận request từ trình duyệt
  • Đọc tham số (parameter) từ request
  • Gọi Model (DAO) để lấy hoặc lưu dữ liệu
  • Chọn View (JSP) để hiển thị
  • Gửi dữ liệu và chuyển hướng đến View

5.2. Cấu trúc một Controller chuẩn

Java - Cấu trúc Controller
package controller;

import dao.TenDAO;
import model.TenModel;
import java.io.IOException;
import java.util.List;
import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

@WebServlet("/ten-duong-dan")
public class TenController extends HttpServlet {
    
    private TenDAO tenDAO = new TenDAO();
    
    @Override
    protected void doGet(HttpServletRequest req, HttpServletResponse resp)
        throws ServletException, IOException {
        
        String action = req.getParameter("action");
        
        if (action == null) {
            action = "list";
        }
        
        switch (action) {
            case "list": list(req, resp); break;
            case "add": showAddForm(req, resp); break;
            case "edit": showEditForm(req, resp); break;
            case "delete": delete(req, resp); break;
            default: list(req, resp);
        }
    }
    
    @Override
    protected void doPost(HttpServletRequest req, HttpServletResponse resp)
        throws ServletException, IOException {
        
        String action = req.getParameter("action");
        
        if ("add".equals(action)) {
            insert(req, resp);
        } else if ("update".equals(action)) {
            update(req, resp);
        } else {
            list(req, resp);
        }
    }
}

6.4. Bước 2: Model - Student.java

model/Student.java
package model;

import java.sql.Timestamp;

public class Student {
    private int id;
    private String studentCode;
    private String fullName;
    private String email;
    private String phone;
    private String major;
    private double gpa;
    private Timestamp createdAt;
    
    public Student() {}
    
    public Student(String studentCode, String fullName, String email,
            String phone, String major, double gpa) {
        this.studentCode = studentCode;
        this.fullName = fullName;
        this.email = email;
        this.phone = phone;
        this.major = major;
        this.gpa = gpa;
    }
    
    // Getter và Setter (đã được viết đầy đủ)
    public int getId() { return id; }
    public void setId(int id) { this.id = id; }
    public String getStudentCode() { return studentCode; }
    public void setStudentCode(String studentCode) { this.studentCode = studentCode; }
    public String getFullName() { return fullName; }
    public void setFullName(String fullName) { this.fullName = fullName; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }
    public String getPhone() { return phone; }
    public void setPhone(String phone) { this.phone = phone; }
    public String getMajor() { return major; }
    public void setMajor(String major) { this.major = major; }
    public double getGpa() { return gpa; }
    public void setGpa(double gpa) { this.gpa = gpa; }
    public Timestamp getCreatedAt() { return createdAt; }
    public void setCreatedAt(Timestamp createdAt) { this.createdAt = createdAt; }
    
    // Phương thức tiện ích: xếp loại học lực theo GPA
    public String getAcademicRank() {
        if (gpa >= 3.6) return "Xuất sắc";
        if (gpa >= 3.2) return "Giỏi";
        if (gpa >= 2.5) return "Khá";
        if (gpa >= 2.0) return "Trung bình";
        return "Yếu";
    }
}
